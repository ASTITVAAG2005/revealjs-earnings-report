Reconciliation Snippet (Code Highlight)

import pandas as pd

def reconcile(ledger_df, gl_df):
    merged = ledger_df.merge(gl_df, on=["account", "period"], suffixes=("_ledger", "_gl"))
    merged["variance"] = merged["amount_ledger"] - merged["amount_gl"]
    return merged.loc[merged["variance"].abs() > 0.01]
